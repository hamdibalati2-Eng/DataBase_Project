CREATE TYPE user_role AS ENUM ('Player', 'HeadCoach', 'Doctor', 'Manager');
CREATE TYPE tournament_status AS ENUM ('Upcoming', 'Ongoing', 'Completed');
CREATE TYPE jersey_enum AS ENUM ('Home', 'Away');
CREATE TYPE session_enum AS ENUM ('Physical', 'Rest', 'Recovery');


CREATE TABLE Users (
    user_id INT PRIMARY KEY,
    name VARCHAR(20),
    password VARCHAR(25),
    role user_role
);

CREATE TABLE Contract (
    Contract_id INT PRIMARY KEY,
    user_id INT,
    start_date DATE NOT NULL,
    end_date DATE NOT NULL,
    salary DECIMAL(10,2) NOT NULL,
    FOREIGN KEY (user_id) REFERENCES Users(user_id) ON DELETE CASCADE
);


CREATE TABLE Players (
    player_id INT PRIMARY KEY,
    Age INT,
    Healthstatus VARCHAR(15),
    Position VARCHAR(10),
    Nationality VARCHAR(15),
    Contract_id INT,
    Jerseynumber INT,
    FOREIGN KEY (player_id) REFERENCES Users(user_id) ON DELETE CASCADE,
    FOREIGN KEY (Contract_id) REFERENCES Contract(Contract_id) ON DELETE CASCADE
);


CREATE TABLE HeadCoaches (
    coach_id INT PRIMARY KEY,
    Contract_id INT,
    Age INT,
    Nationality VARCHAR(20),
    FOREIGN KEY (coach_id) REFERENCES Users(user_id) ON DELETE CASCADE,
    FOREIGN KEY (Contract_id) REFERENCES Contract(Contract_id) ON DELETE CASCADE
);


CREATE TABLE Doctors (
    doctor_id INT PRIMARY KEY,
    Specialization VARCHAR(20),
    Contract_id INT,
    Age INT,
    FOREIGN KEY (doctor_id) REFERENCES Users(user_id) ON DELETE CASCADE,
    FOREIGN KEY (Contract_id) REFERENCES Contract(Contract_id) ON DELETE CASCADE
);


CREATE TABLE Managers (
    manager_id INT PRIMARY KEY,
    FOREIGN KEY (manager_id) REFERENCES Users(user_id) ON DELETE CASCADE
);


CREATE TABLE Tournaments (
    tournament_id INT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    start_date DATE NOT NULL,
    end_date DATE NOT NULL,
    status tournament_status DEFAULT 'Upcoming'
);


CREATE TABLE GamesSchedule (
    game_id INT PRIMARY KEY GENERATED ALWAYS AS IDENTITY,
    tournament_id INT,
    match_date DATE,
    tactics TEXT,
    stadium VARCHAR(100),
    jersey_type jersey_enum,
    FOREIGN KEY (tournament_id) REFERENCES Tournaments(tournament_id) ON DELETE SET NULL
);


CREATE TABLE TrainingSessions (
    session_id INT PRIMARY KEY,
    session_type session_enum NOT NULL,
    session_date DATE NOT NULL
);

CREATE TABLE Achievements (
    achievement_id INT PRIMARY KEY,
    title VARCHAR(100) NOT NULL,
    description TEXT,
    achievement_date DATE NOT NULL
);


CREATE TABLE injuries (
    injury_id INTEGER NOT NULL PRIMARY KEY;
    player_id INTEGER NOT NULL,
    injury_date DATE NOT NULL,
    health_status CHARACTER VARYING(50),
    recovery_date DATE,
    injury_description TEXT
);
